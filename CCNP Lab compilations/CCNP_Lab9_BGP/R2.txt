R2#sh run
!
hostname R2
!
boot-start-marker
boot-end-marker
!
!
!
no aaa new-model
!
memory-size iomem 10
!
no ipv6 cef
ip source-route
ip cef
!
!
!
!
interface GigabitEthernet0/0
 ip address 192.168.2.1 255.255.255.0
 duplex auto
 speed auto
 no shutdown
!
interface GigabitEthernet0/1
 ip address 192.168.3.2 255.255.255.0
 duplex auto
 speed auto
 no shutdown
!
router ospf 1
 redistribute bgp 1
 network 192.168.2.0 0.0.0.255 area 0
 default-information originate
!
router bgp 1
 bgp log-neighbor-changes
 network 192.168.3.0
 redistribute ospf 1
 neighbor 192.168.2.2 remote-as 6
 neighbor 192.168.3.1 remote-as 4
 neighbor 192.168.3.1 next-hop-self
!
!
!
line con 0
line aux 0
line 2
 no activation-character
 no exec
 transport preferred none
 transport input all
 transport output pad telnet rlogin lapb-ta mop udptn v120 ssh
 stopbits 1
line vty 0 4
 login
 transport input all
!
scheduler allocate 20000 1000
end
